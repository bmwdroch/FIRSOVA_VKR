# Структура проекта прогнозирования оттока клиентов

```
telco_churn_prediction/
├── data/
│   ├── raw/                    # Исходные необработанные данные
│   ├── interim/                # Промежуточные данные после трансформаций
│   └── processed/              # Финальные обработанные данные для ML
├── models/                     # Сохраненные ML модели и препроцессоры
├── src/                        # Исходный код проекта
│   ├── __init__.py
│   ├── data/                   # Скрипты для работы с данными
│   │   ├── __init__.py
│   │   ├── download.py         # Загрузка датасета
│   │   └── preprocess.py       # Предобработка данных
│   ├── features/               # Скрипты для создания и трансформации признаков
│   │   ├── __init__.py
│   │   └── build_features.py   # Создание признаков
│   ├── models/                 # Скрипты для обучения и оценки моделей
│   │   ├── __init__.py
│   │   ├── train.py            # Обучение моделей
│   │   ├── predict.py          # Получение предсказаний
│   │   └── evaluate.py         # Оценка моделей
│   └── visualization/          # Скрипты для визуализации
│       ├── __init__.py
│       └── visualize.py        # Создание графиков
├── app/                        # Приложение для прогнозирования
│   ├── __init__.py
│   ├── model_predictor.py      # Основной скрипт приложения
│   └── utils/                  # Вспомогательные функции для приложения
│       ├── __init__.py
│       └── io_utils.py         # Функции ввода-вывода
├── notebooks/                  # Анализ и исследование в структурированном виде
│   ├── 01_data_exploration.py  # Исследовательский анализ
│   ├── 02_model_comparison.py  # Сравнение моделей
│   └── 03_results_analysis.py  # Анализ результатов
├── docs/                       # Документация проекта
│   ├── data_dictionary.md      # Описание набора данных
│   └── model_report.md         # Отчет о моделях
├── tests/                      # Тесты
│   ├── __init__.py
│   ├── test_data.py            # Тесты для работы с данными
│   └── test_models.py          # Тесты для моделей
├── .gitignore                  # Файлы, игнорируемые Git
├── requirements.txt            # Зависимости проекта
├── setup.py                    # Установка проекта как пакета
└── README.md                   # Описание проекта
```
